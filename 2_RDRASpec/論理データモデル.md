```mermaid
        string 名前
        string 連絡先
        string 住所
        string 生年月日
        string 性別
        string 介護度
        enum 介護会員の状況分類
        string 要望内容
        string 契約日
        enum 介護会員状態
        number 個人情報_ID FK
    }

    個人情報 {
        number 個人情報_ID PK
        string 緊急連絡先
        string 家族情報
        string 既往歴
        string アレルギー情報
        string 保険証番号
    }

    サービススタッフ {
        number サービススタッフ_ID PK
        string 名前
        string 連絡先
        number 働き方分類_ID FK
        string 勤務可能日時
        string 要望内容
        enum サービススタッフ状態
        number 事業所_ID FK
    }

    スキル {
        number スキル_ID PK
        string スキル名

        number スキル_ID FK

    }

    働き方分類 {

        string 分類名
        string 勤務形態
        string 勤務時間帯
        string 休日パターン
    }

    介護施設 {
        number 介護施設_ID PK
        string 施設名
        enum 介護施設分類
        string 住所
        string 連絡先
        number 事業所_ID FK
        enum 介護施設状態
    }

    事業所 {
        number 事業所_ID PK
        string 事業所名
        enum 事業所タイプ
        string 住所
        string 連絡先
        number 管理者ID
        number 本部ID
        enum 事業所状態
    }

    訪問介護計画 {

        number 介護会員_ID FK
        number サービススタッフ_ID FK
        string 訪問日時
        string 訪問内容
        number 所要時間
        enum サービス種別
        number 事業所_ID FK
        enum 訪問介護計画状態
    }

    勤務スケジュール {

        number サービススタッフ_ID FK
        string 勤務日
        string 勤務開始時刻
        string 勤務終了時刻
        string 勤務場所
        enum 勤務スケジュール状態
    }

    訪問予定 {
        number 訪問予定_ID PK
        number 訪問介護計画_ID FK
        number 勤務スケジュール_ID FK
        enum 訪問予定状態
    }

    訪問介護記録 {
        number 訪問介護記録_ID PK
        number 訪問介護計画_ID FK
        number 介護会員_ID FK
        number サービススタッフ_ID FK
        string 実施日時
        string 実施内容
        number 実施時間
        string 特記事項
        enum 訪問介護記録状態
    }

    介護費用 {
        number 介護費用_ID PK
        number 介護会員_ID FK
        number 訪問介護記録_ID FK
        number 計算金額
        number 請求金額
        string 請求日
        string 回収日
        string 回収状況
        enum 請求状態
    }

    請求書 {
        number 請求書_ID PK
        number 介護費用_ID FK
        number 介護会員_ID FK
        string 請求先
        enum 請求先区分
        number 請求金額
        string 請求日
        string 支払期限
    }

    回収記録 {
        number 回収記録_ID PK
        number 介護費用_ID FK
        number 回収金額
        string 回収日
        string 回収方法
        number 未回収金額
    }

    法規情報 {
        number 法規情報_ID PK
        string 法規名
        string 適用条件
        string 制約内容
        string 更新日
    }
```
